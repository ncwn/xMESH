\begin{center}
    \fontsize{14}{17}\selectfont{\textbf{LIST OF SYMBOLS}}
\end{center}

\vspace{3em}

\begin{longtable}[l]{ll ll}

$\alpha$           & Learning rate/weighting factor (EWMA smoothing)                   \\ [0.1cm]
$C$                & Composite routing cost (lower is better)                          \\ [0.1cm]
$C_n$              & Cost via neighbor $n$                                             \\ [0.1cm]
$E$ or $ETX$       & Expected Transmission Count                                       \\ [0.1cm]
$h$                & Number of hops to destination                                     \\ [0.1cm]
$I$                & Trickle interval (seconds)                                        \\ [0.1cm]
$I_{min}$          & Minimum Trickle interval (60 seconds)                             \\ [0.1cm]
$I_{max}$          & Maximum Trickle interval (600 seconds)                            \\ [0.1cm]
$N$                & Number of nodes in network                                        \\ [0.1cm]
$P_{weak}$         & Weak link penalty (0 or 1.5)                                      \\ [0.1cm]
$R$ or $RSSI$      & Received Signal Strength Indicator (dBm)                          \\ [0.1cm]
$R_{norm}$         & Normalized RSSI value (0 to 1)                                    \\ [0.1cm]
$S$ or $SNR$       & Signal-to-Noise Ratio (dB)                                        \\ [0.1cm]
$S_{norm}$         & Normalized SNR value (0 to 1)                                     \\ [0.1cm]
$W_1, W_2, W_3, W_4, W_5$ & Weight factors (1.0, 0.3, 0.2, 0.4, 1.0)               \\ [0.1cm]
$b$                & Gateway load bias term                                            \\ [0.1cm]
$\mu$              & Mean value (statistical notation)                                 \\ [0.1cm]
$\sigma$           & Standard deviation (statistical notation)                         \\ [0.1cm]
$\Sigma$           & Summation operator                                                \\ [0.1cm]
$\times$           & Multiplication operator or improvement factor                     \\ [0.1cm]
$\approx$          & Approximately equal to                                            \\ [0.1cm]
$\neq$             & Not equal to                                                      \\ [0.1cm]
$\leq$             & Less than or equal to                                             \\ [0.1cm]
$\geq$             & Greater than or equal to                                          \\ [0.1cm]
$\rightarrow$      & Implies or transition to                                          \\ [0.1cm]
$\pm$              & Plus or minus (confidence interval notation)                      \\ [0.1cm]

\end{longtable}
